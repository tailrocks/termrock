## Usage

```rust
use termrock::{Theme, widgets::{TextArea, TextAreaState, TextCursor}};

let theme = Theme::default();
let area = TextArea::new(&theme).title("Notes").placeholder("Write…");
let mut state = TextAreaState::new("first line
second line");
state.set_focused(true);
let valid = state.set_cursor(TextCursor { line: 1, byte: 0 });
```
