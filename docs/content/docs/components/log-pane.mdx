## Usage

```rust
use termrock::{Theme, widgets::{LogPane, LogPaneState}};

let theme = Theme::default();
let pane = LogPane::new(&theme).title("Build");
let mut state = LogPaneState::new().with_max_lines(1_000);
state.append("compiling termrock");
```
